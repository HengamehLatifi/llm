/** @module HuggingFace */

//This code defines a setup in JavaScript to execute shell commands and interact with a Python script for using a Hugging Face machine learning model. 
//It includes utilities for handling file paths and ensuring compatibility with different operating systems 


import child_process from "child_process"
import { promisify } from "util"
/**
 * excutes a shell command and returns a `promise` that resolves with the result.
 * @function
 * @returns {promise}
 */
const execPromise = promisify(child_process.exec)
import { escapeShell, concatPath } from "../../utils.js"

// directory of this file
/**Creates a URL object representing the directory of the current module to the current directory, and provides the URL of the current module as a file
 * and extracts the path of the URL.
 */
let __dirname = new URL(".", import.meta.url).pathname

// if windows

/**
 *  This block checks if the code is running on a Windows operating system.
 * @type {boolean}
 */
const isWindows = process.platform === "win32"    
if (isWindows) {
  // /C:/ -> /c/

  /**
 * a direct operation on a variable to search within the directory name string for the first capital letter ([A-Z]) that appears at the beginning of the string (/^).
 * @param {string} __dirname
 * @returns {string}
 */
  __dirname = __dirname.replace(/^\/([A-Z]):/, (match, p1) => {
    return "/" + p1.toLowerCase()
  })
}

/**
 * is an asynchronous function designed to interact with a Hugging Face model by invoking a Python script.
 * @type {function(Object): Promise<string>}
 */
export const useHuggingface = async ({ print, args }) => {
  if (!args.quiet) print(args.prompt.gray)
  try {
    const { stdout } = await execPromise(
      `python ${concatPath(
        __dirname,
        "/models/huggingface/use.py"
      )} --model "${escapeShell(args.model)}" --prompt "${escapeShell(
        args.prompt
      )}"`,
      {
        cwd: __dirname,
        encoding: "utf8",
      }
    )
    process.stdout.write(stdout)
    return stdout
  } catch (error) {
    console.error(`Error: ${error.message}`)
    process.exit(1)
  }
}
